import pygame
import sys
from PIL import Image

def load_gif_frames(gif_path):
    """Load GIF frames and convert them to Pygame surfaces."""
    frames = []
    gif = Image.open(gif_path)
    try:
        while True:
            frame = gif.copy()
            frame = frame.convert("RGBA")
            frame_data = frame.tobytes()
            pygame_frame = pygame.image.fromstring(frame_data, frame.size, "RGBA")
            frames.append(pygame_frame)
            gif.seek(gif.tell() + 1)
    except EOFError:
        pass
    return frames

def main():
    pygame.init()

    screen = pygame.display.set_mode((500, 500))
    pygame.display.set_caption("GIF Animation with Pygame")

    # Load GIF frames
    gif_frames = load_gif_frames("path_to_your_gif.gif")
    num_frames = len(gif_frames)
    frame_index = 0
    clock = pygame.time.Clock()

    running = True
    while running:
        for event in pygame.event.get():
            if event.type == pygame.QUIT:
                running = False

        screen.fill((255, 255, 255))  # Fill the screen with white
        screen.blit(gif_frames[frame_index], (0, 0))  # Draw the current frame

        frame_index = (frame_index + 1) % num_frames  # Move to the next frame
        pygame.display.flip()  # Update the display

        clock.tick(10)  # Control the frame rate (10 FPS in this case)

    pygame.quit()
    sys.exit()

if __name__ == "__main__":
    main()
