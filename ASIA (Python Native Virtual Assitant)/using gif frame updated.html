import pygame
import sys
import threading
from PIL import Image

def load_gif_frames(gif_path):
    """Load GIF frames and convert them to Pygame surfaces."""
    frames = []
    gif = Image.open(gif_path)
    try:
        while True:
            frame = gif.copy()
            frame = frame.convert("RGBA")
            mode = frame.mode
            size = frame.size
            data = frame.tobytes()
            pygame_frame = pygame.image.fromstring(data, size, mode)
            frames.append(pygame_frame)
            gif.seek(gif.tell() + 1)
    except EOFError:
        pass
    return frames

def pygame_animation():
    pygame.init()

    # Set the display to full screen
    screen = pygame.display.set_mode((0, 0), pygame.FULLSCREEN)
    pygame.display.set_caption("GIF Animation with Pygame")

    # Load GIF frames
    gif_frames = load_gif_frames("path_to_your_gif.gif")
    num_frames = len(gif_frames)
    frame_index = 0
    clock = pygame.time.Clock()

    running = True
    while running:
        for event in pygame.event.get():
            if event.type == pygame.QUIT:
                running = False
            elif event.type == pygame.KEYDOWN:
                if event.key == pygame.K_ESCAPE:
                    running = False

        screen.fill((255, 255, 255))  # Fill the screen with white
        screen.blit(gif_frames[frame_index], (0, 0))  # Draw the current frame

        frame_index = (frame_index + 1) % num_frames  # Move to the next frame
        pygame.display.flip()  # Update the display

        clock.tick(10)  # Control the frame rate (10 FPS in this case)

    pygame.quit()
    sys.exit()

if __name__ == "__main__":
    # Start the Pygame animation in a separate thread
    animation_thread = threading.Thread(target=pygame_animation)
    animation_thread.start()

    # Main application code continues here
    # For example, a simple loop that waits for the animation to finish
    try:
        while animation_thread.is_alive():
            animation_thread.join(1)  # Wait for 1 second or until the thread finishes
    except KeyboardInterrupt:
        pygame.quit()
        sys.exit()
