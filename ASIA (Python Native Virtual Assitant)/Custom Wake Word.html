import pvporcupine
import pyaudio
import struct

# Replace 'path/to/your_custom_wake_word.ppn' with the actual path to your custom wake word file
custom_keyword_path = 'path/to/your_custom_wake_word.ppn'

# Initialize Porcupine with your custom keyword
porcupine = pvporcupine.create(
    access_key='YOUR_PICOVOICE_ACCESS_KEY',
    keyword_paths=[custom_keyword_path]
)

def audio_callback(in_data, frame_count, time_info, status):
    pcm = struct.unpack_from("h" * porcupine.frame_length, in_data)

    keyword_index = porcupine.process(pcm)
    if keyword_index >= 0:
        print("Wake word detected!")

    return (in_data, pyaudio.paContinue)

pa = pyaudio.PyAudio()
audio_stream = pa.open(
    rate=porcupine.sample_rate,
    channels=1,
    format=pyaudio.paInt16,
    input=True,
    frames_per_buffer=porcupine.frame_length,
    stream_callback=audio_callback
)

print("Listening for wake word...")

try:
    audio_stream.start_stream()
    while True:
        pass
except KeyboardInterrupt:
    audio_stream.stop_stream()
    audio_stream.close()
finally:
    pa.terminate()
    porcupine.delete()
