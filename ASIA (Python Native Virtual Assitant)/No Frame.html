import tkinter as tk
from tkinter import ttk

def create_gif_gui(gif_path):
    root = tk.Tk()
    root.title("GIF Animation")

    # Create a label to display the GIF
    lbl = ttk.Label(root)
    lbl.pack()

    # Load the GIF
    gif = tk.PhotoImage(file=gif_path)

    # Update the image on the label
    def update_frame(frame_num):
        frame_num = (frame_num + 1) % gif.n_frames
        gif.configure(format=f"gif -index {frame_num}")
        lbl.configure(image=gif)
        root.after(100, update_frame, frame_num)

    # Start the animation
    update_frame(0)

    root.mainloop()

if __name__ == "__main__":
    gif_path = "fake_path/to/your_gif.gif"
    create_gif_gui(gif_path)
